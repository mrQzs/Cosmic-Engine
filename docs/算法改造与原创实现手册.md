**步骤 2**: 实现环带分配核心算法

```go
// backend/internal/physics/orbit_allocator.go

func AllocateOrbit(existingSatellites []OrbitalParams, cfg RingConfig, rng *rand.Rand) OrbitalParams {

    // 1. 按 RingIndex 分组统计每层的卫星
    rings := groupByRing(existingSatellites)

    // 2. 扫描每一层，找到"最宽空隙"
    bestRing := -1
    bestGap := 0.0
    bestInsertPhase := 0.0

    for ringIdx := 0; ringIdx < len(rings)+1; ringIdx++ {
        sats := rings[ringIdx]
        if len(sats) >= cfg.MaxPerRing {
            continue // 这层满了
        }
        if len(sats) == 0 {
            // 空层 → 直接使用，相位随机
            bestRing = ringIdx
            bestGap = 2 * math.Pi
            bestInsertPhase = rng.Float64() * 2 * math.Pi
            break
        }

        // 排序该层所有卫星的相位角
        phases := extractAndSortPhases(sats)

        // 找最大空隙
        maxGap, insertPhase := findLargestGap(phases)
        if maxGap > bestGap && maxGap > cfg.MinSafeGap {
            bestGap = maxGap
            bestRing = ringIdx
            bestInsertPhase = insertPhase
        }
    }

    // 3. 如果没有合适的层 → 新开一层
    if bestRing == -1 {
        bestRing = len(rings)
        bestInsertPhase = rng.Float64() * 2 * math.Pi
    }

    // 4. 计算物理参数
    radius := cfg.MinRadius + float64(bestRing)*cfg.RadiusIncrement
    radius *= 1.0 + (rng.Float64()-0.5)*0.2 // ±10% 抖动

    inclination := float64(bestRing) * cfg.InclinationStep
    if bestRing%2 == 1 {
        inclination = -inclination // 奇数层反向倾斜
    }

    return OrbitalParams{
        RingIndex:        bestRing,
        OrbitRadius:      radius,
        OrbitInclination: inclination,
        PhaseOffset:      bestInsertPhase,
        Eccentricity:     0.02 + rng.Float64()*0.06,
        OrbitalSpeed:     1.0 / math.Sqrt(radius*radius*radius) * 50, // 开普勒第三定律
    }
}

// findLargestGap 在已排序的相位角列表中找到最大空隙，返回空隙大小和中点
func findLargestGap(sortedPhases []float64) (float64, float64) {
    n := len(sortedPhases)
    maxGap := 0.0
    insertPhase := 0.0

    for i := 0; i < n; i++ {
        next := sortedPhases[(i+1)%n]
        curr := sortedPhases[i]
        gap := next - curr
        if i == n-1 {
            gap = (2*math.Pi - curr) + next // 跨越 0/2π 的间隙
        }
        if gap > maxGap {
            maxGap = gap
            insertPhase = curr + gap/2
            if insertPhase >= 2*math.Pi {
                insertPhase -= 2 * math.Pi
            }
        }
    }
    return maxGap, insertPhase
}
```

**步骤 1**: 实现开普勒方程求解器

```ts
// frontend/utils/kepler.ts

/**
 * Newton-Raphson 迭代求解开普勒方程: M = E - e * sin(E)
 * @param M 平均近点角 (mean anomaly)
 * @param e 离心率 (0 < e < 1)
 * @returns E 偏近点角 (eccentric anomaly)
 */
export function solveKepler(M: number, e: number): number {
  let E = M // 初始猜测
  for (let i = 0; i < 3; i++) { // 3 次迭代，e < 0.3 时精度已足够
    const dE = (E - e * Math.sin(E) - M) / (1 - e * Math.cos(E))
    E -= dE
  }
  return E
}

/**
 * 计算椭圆轨道上的 2D 位置（轨道平面内）
 * @param a 半长轴
 * @param e 离心率
 * @param E 偏近点角
 * @returns [x, z] 轨道平面内的坐标
 */
export function ellipsePosition(a: number, e: number, E: number): [number, number] {
  const x = a * (Math.cos(E) - e)
  const z = a * Math.sqrt(1 - e * e) * Math.sin(E)
  return [x, z]
}

/**
 * 根据轨道参数和时间计算天体 3D 位置
 */
export function computeOrbitalPosition(
  time: number,
  params: {
    semiMajorAxis: number   // 半长轴 a
    eccentricity: number    // 离心率 e (0.05~0.3)
    inclination: number     // 倾角（弧度）
    phaseOffset: number     // 初始相位偏移
    orbitalSpeed: number    // 公转角速度
  }
): [number, number, number] {
  const { semiMajorAxis: a, eccentricity: e, inclination, phaseOffset, orbitalSpeed } = params

  // 1. 平均近点角
  const M = time * orbitalSpeed + phaseOffset

  // 2. 求解开普勒方程得偏近点角
  const E = solveKepler(M, e)

  // 3. 轨道平面内的 2D 位置
  const [x2d, z2d] = ellipsePosition(a, e, E)

  // 4. 应用轨道倾角旋转到 3D
  const x = x2d
  const y = z2d * Math.sin(inclination)
  const z = z2d * Math.cos(inclination)

  return [x, y, z]
}
```

```ts
import * as THREE from 'three'
import { solveKepler, ellipsePosition } from './kepler'

export function createEllipseOrbitLine(
  a: number, e: number, inclination: number, segments = 64
): THREE.BufferGeometry {
  const points: THREE.Vector3[] = []

  for (let i = 0; i <= segments; i++) {
    const E = (i / segments) * Math.PI * 2
    const [x2d, z2d] = ellipsePosition(a, e, E)
    const x = x2d
    const y = z2d * Math.sin(inclination)
    const z = z2d * Math.cos(inclination)
    points.push(new THREE.Vector3(x, y, z))
  }

  return new THREE.BufferGeometry().setFromPoints(points)
}
```

```go
// backend/internal/relation/jaccard.go

func JaccardSimilarity(tagsA, tagsB []string) float64 {
    if len(tagsA) == 0 && len(tagsB) == 0 {
        return 0
    }

    setA := make(map[string]bool, len(tagsA))
    for _, t := range tagsA {
        setA[t] = true
    }

    intersection := 0
    setB := make(map[string]bool, len(tagsB))
    for _, t := range tagsB {
        setB[t] = true
        if setA[t] {
            intersection++
        }
    }

    union := len(setA)
    for t := range setB {
        if !setA[t] {
            union++
        }
    }

    return float64(intersection) / float64(union)
}
```

**步骤 3**: 实现综合关联度计算

```go
// backend/internal/relation/similarity.go

type RelationScore struct {
    BodyIDA   uuid.UUID `json:"body_id_a"`
    BodyIDB   uuid.UUID `json:"body_id_b"`
    Score     float64   `json:"score"`
    Reasons   []string  `json:"reasons"` // 供前端 tooltip 展示
}

const (
    WeightJaccard  = 0.4
    WeightVector   = 0.5
    WeightGalaxy   = 0.1
    Threshold      = 0.35
)

func ComputeRelations(ctx context.Context, db *pgxpool.Pool, bodyID uuid.UUID) ([]RelationScore, error) {
    var results []RelationScore

    // 1. 获取当前文章
    body, err := fetchBody(ctx, db, bodyID)
    if err != nil {
        return nil, err
    }

    // 2. pgvector 余弦相似度 — 取 Top 20 最相似
    rows, err := db.Query(ctx, `
        SELECT id, title, tags, galaxy_id,
               1 - (embedding <=> $1) AS cosine_sim
        FROM celestial_bodies
        WHERE type = 'PLANET'
          AND id != $2
          AND deleted_at IS NULL
          AND embedding IS NOT NULL
        ORDER BY embedding <=> $1
        LIMIT 20
    `, body.Embedding, bodyID)
    if err != nil {
        return nil, err
    }
    defer rows.Close()

    for rows.Next() {
        var other struct {
            ID        uuid.UUID
            Title     string
            Tags      []string
            GalaxyID  uuid.UUID
            CosineSim float64
        }
        if err := rows.Scan(&other.ID, &other.Title, &other.Tags, &other.GalaxyID, &other.CosineSim); err != nil {
            continue
        }

        // 3. Jaccard 标签相似度
        jaccard := JaccardSimilarity(body.Tags, other.Tags)

        // 4. 同星系加分
        galaxyBonus := 0.0
        if body.GalaxyID == other.GalaxyID {
            galaxyBonus = 1.0
        }

        // 5. 加权求和
        score := WeightVector*other.CosineSim + WeightJaccard*jaccard + WeightGalaxy*galaxyBonus

        if score >= Threshold {
            reasons := []string{}
            if jaccard > 0 {
                reasons = append(reasons, fmt.Sprintf("共同标签 (%.0f%%)", jaccard*100))
            }
            if other.CosineSim > 0.5 {
                reasons = append(reasons, fmt.Sprintf("内容相似 (%.0f%%)", other.CosineSim*100))
            }
            if galaxyBonus > 0 {
                reasons = append(reasons, "同一星系")
            }
            results = append(results, RelationScore{
                BodyIDA: bodyID,
                BodyIDB: other.ID,
                Score:   score,
                Reasons: reasons,
            })
        }
    }
    return results, nil
}
```

```go
// backend/internal/crypto/encrypt.go

import (
    "crypto/aes"
    "crypto/cipher"
    "crypto/rand"
    "golang.org/x/crypto/argon2"
)

type EncryptedPayload struct {
    Ciphertext []byte `json:"ciphertext"` // 密文（含 AuthTag）
    IV         []byte `json:"iv"`         // 12 字节初始化向量
    Salt       []byte `json:"salt"`       // 16 字节盐值
}

func Encrypt(plaintext []byte, passphrase string) (*EncryptedPayload, error) {
    // 1. 生成随机 salt
    salt := make([]byte, 16)
    if _, err := rand.Read(salt); err != nil {
        return nil, err
    }

    // 2. Argon2id 密钥派生：口令 → 256 位密钥
    key := argon2.IDKey([]byte(passphrase), salt, 3, 64*1024, 4, 32)

    // 3. AES-256-GCM 加密
    block, err := aes.NewCipher(key)
    if err != nil {
        return nil, err
    }
    gcm, err := cipher.NewGCM(block)
    if err != nil {
        return nil, err
    }

    // 4. 生成随机 IV
    iv := make([]byte, gcm.NonceSize()) // 12 bytes
    if _, err := rand.Read(iv); err != nil {
        return nil, err
    }

    // 5. 加密（Seal 会将 AuthTag 追加到密文末尾）
    ciphertext := gcm.Seal(nil, iv, plaintext, nil)

    return &EncryptedPayload{
        Ciphertext: ciphertext,
        IV:         iv,
        Salt:       salt,
    }, nil
}
```

**步骤 2**: 前端解密流程（Web Crypto API）

```ts
// frontend/utils/crypto.ts

export async function decryptArticle(
  passphrase: string,
  payload: { ciphertext: string; iv: string; salt: string } // Base64 编码
): Promise<string> {
  const encoder = new TextEncoder()
  const decoder = new TextDecoder()

  // 1. Base64 解码
  const ciphertextBuf = base64ToBuffer(payload.ciphertext)
  const ivBuf = base64ToBuffer(payload.iv)
  const saltBuf = base64ToBuffer(payload.salt)

  // 2. PBKDF2 密钥派生（Web Crypto API 不支持 Argon2，用 PBKDF2 替代）
  //    600,000 次 SHA-256 迭代 ≈ 200-500ms
  const keyMaterial = await crypto.subtle.importKey(
    'raw',
    encoder.encode(passphrase),
    'PBKDF2',
    false,
    ['deriveKey']
  )

  const key = await crypto.subtle.deriveKey(
    {
      name: 'PBKDF2',
      salt: saltBuf,
      iterations: 600_000,
      hash: 'SHA-256',
    },
    keyMaterial,
    { name: 'AES-GCM', length: 256 },
    false,
    ['decrypt']
  )

  // 3. AES-256-GCM 解密
  try {
    const plainBuf = await crypto.subtle.decrypt(
      { name: 'AES-GCM', iv: ivBuf, tagLength: 128 },
      key,
      ciphertextBuf
    )
    return decoder.decode(plainBuf)
  } catch {
    throw new Error('DECRYPTION_FAILED') // 密码错误或数据损坏
  }
}

function base64ToBuffer(b64: string): ArrayBuffer {
  const binary = atob(b64)
  const buf = new Uint8Array(binary.length)
  for (let i = 0; i < binary.length; i++) buf[i] = binary.charCodeAt(i)
  return buf.buffer
}
```

```tsx
// frontend/components/canvas/CrossFadeLOD.tsx

import { useFrame } from '@react-three/fiber'
import { useRef, useState, useMemo } from 'react'
import * as THREE from 'three'

interface LODLevel {
  distance: number        // 切换距离
  component: React.ReactNode // 该级别的渲染组件
}

interface CrossFadeLODProps {
  levels: LODLevel[]       // 从高到低排列
  transitionWidth?: number // 过渡带宽度比例 (默认 0.15 = ±15%)
  position: THREE.Vector3
}

export function CrossFadeLOD({ levels, transitionWidth = 0.15, position }: CrossFadeLODProps) {
  const [activeLevels, setActiveLevels] = useState<{ idx: number; fade: number }[]>([{ idx: 0, fade: 0 }])
  const prevLevelRef = useRef(0)

  useFrame(({ camera }) => {
    const dist = camera.position.distanceTo(position)

    // 确定当前应显示的 LOD 级别
    let targetLevel = 0
    for (let i = levels.length - 1; i >= 0; i--) {
      if (dist >= levels[i].distance) {
        targetLevel = i
        break
      }
    }

    if (targetLevel !== prevLevelRef.current) {
      const threshold = levels[Math.min(targetLevel, prevLevelRef.current)].distance
      const bandWidth = threshold * transitionWidth

      // 计算过渡进度
      const progress = THREE.MathUtils.clamp(
        (dist - (threshold - bandWidth)) / (2 * bandWidth),
        0, 1
      )

      // 过渡期间同时显示两个级别
      setActiveLevels([
        { idx: prevLevelRef.current, fade: targetLevel > prevLevelRef.current ? progress : 1 - progress },
        { idx: targetLevel, fade: targetLevel > prevLevelRef.current ? 1 - progress : progress },
      ])

      if (progress >= 1 || progress <= 0) {
        prevLevelRef.current = targetLevel
        setActiveLevels([{ idx: targetLevel, fade: 0 }])
      }
    }
  })

  return (
    <group>
      {activeLevels.map(({ idx, fade }) => (
        <DitherFadeWrapper key={idx} fadeProgress={fade}>
          {levels[idx].component}
        </DitherFadeWrapper>
      ))}
    </group>
  )
}
```

```go
        return false // 已过期

    count := 0

        if b == 0 {
            count += 8
            continue
        }
        count += bits.LeadingZeros8(b)
        break
    }
    return count
```

```ts
  challengeId: string,
  difficulty: number

  // 使用 Web Worker 避免阻塞主线程
  return new Promise((resolve) => {

      { type: 'module' }

// frontend/utils/powWorker.ts
self.onmessage = async (e: MessageEvent) => {
  const { challengeId, difficulty } = e.data
  let nonce = 0

  while (true) {
    const data = new TextEncoder().encode(challengeId + nonce.toString())
    const hashBuf = await crypto.subtle.digest('SHA-256', data)
    const hash = new Uint8Array(hashBuf)

    if (countLeadingZeroBits(hash) >= difficulty) {
      self.postMessage({ nonce: nonce.toString() })
      return
    }
    nonce++
  }
}

function countLeadingZeroBits(hash: Uint8Array): number {
  let count = 0
  for (const byte of hash) {
    if (byte === 0) { count += 8; continue }
    count += Math.clz32(byte) - 24 // clz32 counts for 32-bit, byte is 8-bit
    break
  }
  return count
}
```

```go
type ViewCounter struct {
    rdb *redis.Client
}

const bloomThreshold = 100_000 // SET 大小超过此值时迁移到 Bloom Filter

func (vc *ViewCounter) RecordView(ctx context.Context, slug, ipHash string) (bool, error) {
    dateKey := time.Now().Format("2006-01-02")
    setKey := fmt.Sprintf("views:%s:%s", slug, dateKey)
    bloomKey := fmt.Sprintf("views:bf:%s:%s", slug, dateKey)
    modeKey := fmt.Sprintf("views:mode:%s:%s", slug, dateKey)

    // 检查当前使用的是 SET 还是 Bloom Filter
    mode, _ := vc.rdb.Get(ctx, modeKey).Result()

    if mode == "bloom" {
        // Bloom Filter 模式
        exists, err := vc.rdb.Do(ctx, "BF.EXISTS", bloomKey, ipHash).Bool()
        if err != nil || exists {
            return false, err // 已计数
        }
        vc.rdb.Do(ctx, "BF.ADD", bloomKey, ipHash)
        vc.rdb.Incr(ctx, fmt.Sprintf("views:count:%s:%s", slug, dateKey))
        return true, nil
    }

    // SET 模式（默认）
    added, err := vc.rdb.SAdd(ctx, setKey, ipHash).Result()
    if err != nil {
        return false, err
    }
    if added == 0 {
        return false, nil // 已存在
    }

    // 设置 TTL（首次添加时）
    vc.rdb.Expire(ctx, setKey, 25*time.Hour)

    // 检查是否需要迁移到 Bloom Filter
    size, _ := vc.rdb.SCard(ctx, setKey).Result()
    if size >= bloomThreshold {
        go vc.migrateToBloom(context.Background(), slug, dateKey, setKey, bloomKey, modeKey)
    }

    return true, nil
}

func (vc *ViewCounter) GetViewCount(ctx context.Context, slug, date string) (int64, error) {
    modeKey := fmt.Sprintf("views:mode:%s:%s", slug, date)
    mode, _ := vc.rdb.Get(ctx, modeKey).Result()

    if mode == "bloom" {
        return vc.rdb.Get(ctx, fmt.Sprintf("views:count:%s:%s", slug, date)).Int64()
    }
    return vc.rdb.SCard(ctx, fmt.Sprintf("views:%s:%s", slug, date)).Result()
}

func (vc *ViewCounter) migrateToBloom(ctx context.Context, slug, date, setKey, bloomKey, modeKey string) {
    // 1. 创建 Bloom Filter（误判率 1%，预期容量 100 万）
    vc.rdb.Do(ctx, "BF.RESERVE", bloomKey, 0.01, 1_000_000)

    // 2. 将 SET 中的所有元素导入 Bloom Filter
    members, _ := vc.rdb.SMembers(ctx, setKey).Result()
    for _, m := range members {
        vc.rdb.Do(ctx, "BF.ADD", bloomKey, m)
    }

    // 3. 记录当前计数
    vc.rdb.Set(ctx, fmt.Sprintf("views:count:%s:%s", slug, date), len(members), 25*time.Hour)

    // 4. 切换模式标记
    vc.rdb.Set(ctx, modeKey, "bloom", 25*time.Hour)

    // 5. 删除旧 SET 释放内存
    vc.rdb.Del(ctx, setKey)
}
```

```go
// InsertSatellite 在不扰动现有卫星的前提下为新卫星找到最优位置
func InsertSatellite(existing []OrbitalParams, cfg RingConfig, rng *rand.Rand) (
    newSat OrbitalParams,
    adjustments []OrbitalAdjustment, // 需要微调的现有卫星（通常为空）
) {
    // 直接复用 M-1 的 AllocateOrbit
    newSat = AllocateOrbit(existing, cfg, rng)

    // 大多数情况下 adjustments 为空 — 新卫星插入空隙，不影响其他
    // 仅当该层原有卫星被挤压到 MinSafeGap 边缘时，微调相邻两颗
    ringPeers := filterByRing(existing, newSat.RingIndex)
    if len(ringPeers) > 0 {
        adjustments = nudgeNeighbors(ringPeers, newSat.PhaseOffset, cfg.MinSafeGap * 0.8)
    }

    return newSat, adjustments
}

type OrbitalAdjustment struct {
    SatelliteID uuid.UUID     `json:"satellite_id"`
    OldPhase    float64       `json:"old_phase"`
    NewPhase    float64       `json:"new_phase"`
}

// nudgeNeighbors 微调空隙两侧的卫星使空间均匀
func nudgeNeighbors(peers []OrbitalParams, insertPhase, tolerance float64) []OrbitalAdjustment {
    var adjustments []OrbitalAdjustment

    for i, peer := range peers {
        gap := angleDiff(peer.PhaseOffset, insertPhase)
        if gap < tolerance {
            // 需要推开 — 向远离插入点的方向偏移
            nudge := (tolerance - gap) * 0.5 // 仅偏移一半，对称推
            direction := 1.0
            if normalizeAngle(peer.PhaseOffset - insertPhase) < math.Pi {
                direction = -1.0
            }
            newPhase := normalizeAngle(peer.PhaseOffset + nudge*direction)

            adjustments = append(adjustments, OrbitalAdjustment{
                SatelliteID: peers[i].ID,
                OldPhase:    peer.PhaseOffset,
                NewPhase:    newPhase,
            })
        }
    }
    return adjustments
}
```

```go
func RemoveSatellite(existing []OrbitalParams, removeID uuid.UUID) []OrbitalAdjustment {

    var ringPeers []OrbitalParams

        if s.ID == removeID {
```

**步骤 3**: 批量排序切换——匈牙利算法最优指派

```go
// ReorderSatellites 排序切换时最小化全局移动距离
func ReorderSatellites(current []OrbitalParams, targetPositions []OrbitalParams) []OrbitalAdjustment {
    n := len(current)
    // 构建代价矩阵: cost[i][j] = 卫星 i 移动到目标位置 j 的角距离
    cost := make([][]float64, n)
    for i := range cost {
        cost[i] = make([]float64, n)
        for j := range cost[i] {
            cost[i][j] = angleDiff(current[i].PhaseOffset, targetPositions[j].PhaseOffset)
        }
    }

    // 匈牙利算法求解最优指派
    assignment := hungarian(cost)

    var adjustments []OrbitalAdjustment
    for i, j := range assignment {
        if angleDiff(current[i].PhaseOffset, targetPositions[j].PhaseOffset) > 0.01 { // 忽略微小变动
            adjustments = append(adjustments, OrbitalAdjustment{
                SatelliteID: current[i].ID,
                OldPhase:    current[i].PhaseOffset,
                NewPhase:    targetPositions[j].PhaseOffset,
            })
        }
    }

    // 稳定性约束: 最多移动 20% 的卫星
    maxMoves := n / 5
    if len(adjustments) > maxMoves {
        // 按移动距离排序，只保留距离最大的 top-N（移动收益最大的）
        sort.Slice(adjustments, func(i, j int) bool {
            return angleDiff(adjustments[i].OldPhase, adjustments[i].NewPhase) >
                angleDiff(adjustments[j].OldPhase, adjustments[j].NewPhase)
        })
        adjustments = adjustments[:maxMoves]
    }
```

**步骤 2**: 实现注意力加权计算

```ts
// frontend/hooks/useAttentionLOD.ts

import { useFrame, useThree } from '@react-three/fiber'
import * as THREE from 'three'

const WEIGHTS = {
  distance: 0.30,
  gaze:     0.25,
  popular:  0.20,
  recency:  0.10,
  interact: 0.15,
}

// LOD 级别阈值
const LOD_THRESHOLDS = { high: 0.65, mid: 0.35, low: 0.15 }

export function computeAttentionWeight(
  bodyPosition: THREE.Vector3,
  camera: THREE.Camera,
  attention: BodyAttention | undefined,
  navTargetId: string | null,
  bodyId: string
): number {
  const now = Date.now()

  // 1. 距离因子: 距离越近越高 (0~1)
  const dist = camera.position.distanceTo(bodyPosition)
  const distFactor = THREE.MathUtils.clamp(1 - dist / 2000, 0, 1)

  // 2. 视线方向因子: 摄像机朝向与天体方向的点积 (0~1)
  const cameraDir = new THREE.Vector3()
  camera.getWorldDirection(cameraDir)
  const toBody = bodyPosition.clone().sub(camera.position).normalize()
  const gazeFactor = THREE.MathUtils.clamp((cameraDir.dot(toBody) + 1) / 2, 0, 1)

  // 3. 人气因子 (评论数 + 阅读量归一化)
  const comments = attention?.commentCount ?? 0
  const views = attention?.viewCount ?? 0
  const popularFactor = THREE.MathUtils.clamp((comments * 10 + views) / 10000, 0, 1)

  // 4. 新鲜度因子 (7天内的文章满分, 逐渐衰减)
  const ageMs = now - (attention?.publishedAt ?? 0)
  const ageDays = ageMs / (1000 * 60 * 60 * 24)
  const recencyFactor = THREE.MathUtils.clamp(1 - ageDays / 30, 0, 1) // 30天衰减到0

  // 5. 交互因子 (最近 5 秒内悬停/点击过)
  const hoverAge = (now - (attention?.lastHoverTime ?? 0)) / 1000
  const clickAge = (now - (attention?.lastClickTime ?? 0)) / 1000
  const interactFactor = hoverAge < 5 ? 1 : clickAge < 10 ? 0.5 : 0

  // 6. 导航因子 (路由正在指向这个天体)
  const navFactor = navTargetId === bodyId ? 1.0 : 0

  // 加权求和
  let W = WEIGHTS.distance * distFactor
    + WEIGHTS.gaze * gazeFactor
    + WEIGHTS.popular * popularFactor
    + WEIGHTS.recency * recencyFactor
    + WEIGHTS.interact * interactFactor

  // 导航因子直接覆盖（如果正在飞向这个天体，强制最高权重）
  if (navFactor > 0) W = 1.0

  return THREE.MathUtils.clamp(W, 0, 1)
}

export function weightToLODLevel(weight: number): 'high' | 'mid' | 'low' | 'point' {
  if (weight >= LOD_THRESHOLDS.high) return 'high'
  if (weight >= LOD_THRESHOLDS.mid) return 'mid'
  if (weight >= LOD_THRESHOLDS.low) return 'low'
  return 'point'
}
```

**验证方法**:
1. 远处热门文章（100 评论）应渲染 mid LOD，远处冷门文章为 point
2. 悬停 → 移开后 5 秒内保持 LOD 不降
3. 摄像机飞向目标行星时，目标提前升到 high LOD

### I-4. 渐进式场景揭示算法 (Progressive Scene Revelation)

**涉及文件**: `frontend/hooks/useProgressiveReveal.ts`, `frontend/stores/revealStore.ts`
**所属 Phase**: 1（前端骨架）

#### 实现步骤

```ts
// frontend/hooks/useProgressiveReveal.ts

import { useFrame, useThree } from '@react-three/fiber'
import { useRef, useCallback } from 'react'

type RevealPhase = 'skybox' | 'near_galaxies' | 'planets' | 'background' | 'details' | 'complete'

const PHASE_CONFIG: Record<RevealPhase, { deadline: number; next: RevealPhase | null }> = {
  skybox:        { deadline: 500,   next: 'near_galaxies' },
  near_galaxies: { deadline: 1500,  next: 'planets' },
  planets:       { deadline: 3000,  next: 'background' },
  background:    { deadline: 10000, next: 'details' },
  details:       { deadline: 20000, next: 'complete' },
  complete:      { deadline: Infinity, next: null },
}

export function useProgressiveReveal(totalBodies: CelestialBody[], cameraPosition: THREE.Vector3) {
  const phaseRef = useRef<RevealPhase>('skybox')
  const startTime = useRef(Date.now())
  const loadQueueRef = useRef<CelestialBody[]>([])
  const revealedRef = useRef(new Set<string>())
  const isUserActive = useRef(false)

  // 监听用户交互 — 交互时暂停后台加载
  const onInteraction = useCallback(() => { isUserActive.current = true }, [])
  const onIdle = useCallback(() => { isUserActive.current = false }, [])

  useFrame(() => {
    const elapsed = Date.now() - startTime.current
    const phase = phaseRef.current
    const config = PHASE_CONFIG[phase]

    // 检查是否该进入下一阶段
    if (elapsed >= config.deadline && config.next) {
      phaseRef.current = config.next
      onPhaseTransition(config.next, totalBodies, cameraPosition, loadQueueRef)
    }

    // 每帧从队列中取 1 个天体激活（不阻塞渲染）
    if (!isUserActive.current && loadQueueRef.current.length > 0) {
      const body = loadQueueRef.current.shift()!
      revealedRef.current.add(body.id)
    }
  })

  return {
    isRevealed: (bodyId: string) => revealedRef.current.has(bodyId),
    currentPhase: phaseRef.current,
    onInteraction,
    onIdle,
  }
}

function onPhaseTransition(
  newPhase: RevealPhase,
  allBodies: CelestialBody[],
  cameraPos: THREE.Vector3,
  queue: React.MutableRefObject<CelestialBody[]>
) {
  switch (newPhase) {
    case 'near_galaxies': {
      // 按距离排序，取最近 3 个星系
      const galaxies = allBodies
        .filter(b => b.type === 'GALAXY')
        .sort((a, b) => distTo(a, cameraPos) - distTo(b, cameraPos))
        .slice(0, 3)
      queue.current.push(...galaxies)
      break
    }
    case 'planets': {
      // 视野内星系的行星
      const nearPlanets = allBodies
        .filter(b => b.type === 'PLANET' && distTo(b, cameraPos) < 800)
        .sort((a, b) => distTo(a, cameraPos) - distTo(b, cameraPos))
      queue.current.push(...nearPlanets)
      break
    }
    case 'background': {
      // 视野外的星系和特殊天体
      const rest = allBodies.filter(b =>
        !queue.current.includes(b) && (b.type === 'GALAXY' || b.type === 'WORMHOLE')
      )
      queue.current.push(...rest)
      break
    }
    case 'details': {
      // 小行星带、星座连线等
      const details = allBodies.filter(b => b.type === 'ASTEROID')
      queue.current.push(...details)
      break
    }
  }
}
```

**验证方法**:
1. 首屏 500ms 内应仅出现星空背景
2. 1.5s 内最近星系可见
3. 用户交互时暂停后台加载（帧率优先）
4. 直接访问 `/post/slug` 应跳过全景阶段

### I-5. 宇宙布局稳定性维护算法 (Cosmic Layout Stability)

**涉及文件**: `backend/internal/physics/layout_stabilizer.go`
**所属 Phase**: 3-4

#### 实现步骤

```go
// backend/internal/physics/layout_stabilizer.go

type LayoutStabilizer struct {
    db         *pgxpool.Pool
    minDist    float64 // 天体间最小距离
    iterations int     // 斥力迭代次数
}

func NewLayoutStabilizer(db *pgxpool.Pool) *LayoutStabilizer {
    return &LayoutStabilizer{db: db, minDist: 15.0, iterations: 10}
}

// PositionNewArticle 为新文章计算坐标，不移动任何已有天体
func (ls *LayoutStabilizer) PositionNewArticle(
    ctx context.Context,
    articleID uuid.UUID,
    embedding []float32,
    galaxyID uuid.UUID,
) ([3]float64, error) {

    // 1. 查找 Top-3 语义最相似的已有文章坐标
    rows, err := ls.db.Query(ctx, `
        SELECT physics_params->'base_coordinates' AS coords,
               1 - (embedding <=> $1) AS sim
        FROM celestial_bodies
        WHERE type = 'PLANET' AND deleted_at IS NULL
          AND embedding IS NOT NULL AND id != $2
        ORDER BY embedding <=> $1
        LIMIT 3
    `, embedding, articleID)
    if err != nil {
        return [3]float64{}, err
    }
    defer rows.Close()

    type neighbor struct {
        pos [3]float64
        sim float64
    }
    var neighbors []neighbor
    for rows.Next() {
        var n neighbor
        var coordsJSON []byte
        rows.Scan(&coordsJSON, &n.sim)
        json.Unmarshal(coordsJSON, &n.pos)
        neighbors = append(neighbors, n)
    }

    // 2. 计算加权质心
    var pos [3]float64
    var totalWeight float64
    if len(neighbors) == 0 {
        // 无邻居 → 在星系中心附近随机放置
        galaxyCenter, _ := ls.getGalaxyCenter(ctx, galaxyID)
        rng := rand.New(rand.NewSource(time.Now().UnixNano()))
        return [3]float64{
            galaxyCenter[0] + (rng.Float64()-0.5)*100,
            galaxyCenter[1] + (rng.Float64()-0.5)*20,
            galaxyCenter[2] + (rng.Float64()-0.5)*100,
        }, nil
    }
    for _, n := range neighbors {
        w := n.sim * n.sim // 相似度的平方作为权重（放大差异）
        for i := 0; i < 3; i++ {
            pos[i] += n.pos[i] * w
        }
        totalWeight += w
    }
    for i := 0; i < 3; i++ {
        pos[i] /= totalWeight
    }

    // 3. 斥力迭代：推开与现有天体重叠的位置
    allBodies, _ := ls.getAllBodyPositions(ctx)
    for iter := 0; iter < ls.iterations; iter++ {
        for _, body := range allBodies {
            dx := pos[0] - body[0]
            dy := pos[1] - body[1]
            dz := pos[2] - body[2]
            dist := math.Sqrt(dx*dx + dy*dy + dz*dz)

            if dist < ls.minDist && dist > 0.01 {
                // 斥力推开
                force := (ls.minDist - dist) * 0.5 / dist
                pos[0] += dx * force
                pos[1] += dy * force
                pos[2] += dz * force
            }
        }
    }

    // 4. 确保在所属星系范围内（弹性边界）
    galaxyCenter, galaxyRadius := ls.getGalaxyCenterAndRadius(ctx, galaxyID)
    distFromCenter := math.Sqrt(
        math.Pow(pos[0]-galaxyCenter[0], 2) +
        math.Pow(pos[1]-galaxyCenter[1], 2) +
        math.Pow(pos[2]-galaxyCenter[2], 2))

    if distFromCenter > galaxyRadius {
        // 扩展星系边界 10%
        ls.expandGalaxyRadius(ctx, galaxyID, galaxyRadius*1.1)
    }

    return pos, nil
}
```

**验证方法**:
1. 连续添加 20 篇文章，验证每次仅新文章有坐标，旧文章坐标零变化
2. 语义相似的文章（同标签）自然聚集
3. 全部文章间距 >= minDist

### I-6. 摄像机意图预测器 (Camera Intent Predictor)

**涉及文件**: `frontend/hooks/useCameraIntentPredictor.ts`
**所属 Phase**: 8+

#### 实现步骤

```ts
// frontend/hooks/useCameraIntentPredictor.ts

import { useFrame, useThree } from '@react-three/fiber'
import { useRef, useEffect } from 'react'
import * as THREE from 'three'

interface PredictionSignal {
  bodyId: string
  confidence: number  // 0~1
}

const SIGNAL_WEIGHTS = {
  gazeStay:   0.40,  // 视线停留
  mouseTrail: 0.30,  // 鼠标轨迹方向
  context:    0.20,  // 导航上下文
  heatmap:    0.10,  // 全局跳转热力
}

export function useCameraIntentPredictor(
  bodies: CelestialBody[],
  onPrefetch: (bodyId: string) => void  // 触发预加载的回调
) {
  const gazeTimerRef = useRef<Map<string, number>>(new Map()) // bodyId → 累积注视时间(秒)
  const lastPredictionRef = useRef<string | null>(null)
  const mouseDirRef = useRef(new THREE.Vector2())

  // 追踪鼠标方向趋势
  useEffect(() => {
    let prevX = 0, prevY = 0
    const onMouseMove = (e: MouseEvent) => {
      const dx = e.clientX - prevX
      const dy = e.clientY - prevY
      // 指数移动平均
      mouseDirRef.current.x = mouseDirRef.current.x * 0.8 + dx * 0.2
      mouseDirRef.current.y = mouseDirRef.current.y * 0.8 + dy * 0.2
      prevX = e.clientX
      prevY = e.clientY
    }
    window.addEventListener('mousemove', onMouseMove)
    return () => window.removeEventListener('mousemove', onMouseMove)
  }, [])

  useFrame(({ camera }, delta) => {
    const cameraDir = new THREE.Vector3()
    camera.getWorldDirection(cameraDir)

    let bestCandidate: PredictionSignal | null = null

    for (const body of bodies) {
      if (body.type !== 'PLANET') continue

      const bodyPos = new THREE.Vector3(
        body.physics_params.base_coordinates.x,
        body.physics_params.base_coordinates.y,
        body.physics_params.base_coordinates.z,
      )

      // --- 信号 1: 视线停留 ---
      const toBody = bodyPos.clone().sub(camera.position).normalize()
      const gazeAlignment = cameraDir.dot(toBody) // -1 ~ 1
      const isGazing = gazeAlignment > 0.9 // 视线方向 ±25° 以内

      const gazeMap = gazeTimerRef.current
      if (isGazing) {
        gazeMap.set(body.id, (gazeMap.get(body.id) ?? 0) + delta)
      } else {
        gazeMap.set(body.id, Math.max((gazeMap.get(body.id) ?? 0) - delta * 2, 0))
      }
      const gazeTime = gazeMap.get(body.id) ?? 0
      const gazeScore = THREE.MathUtils.clamp(gazeTime / 2, 0, 1) // 2 秒注视 → 满分

      // --- 信号 2: 鼠标轨迹方向 ---
      // 将天体投影到屏幕空间，与鼠标移动方向比较
      const projected = bodyPos.clone().project(camera)
      const screenDir = new THREE.Vector2(projected.x, -projected.y)
      const mouseDir = mouseDirRef.current.clone().normalize()
      const mouseDot = screenDir.normalize().dot(mouseDir)
      const mouseScore = THREE.MathUtils.clamp((mouseDot + 1) / 2, 0, 1)

      // --- 综合置信度 ---
      const confidence =
        SIGNAL_WEIGHTS.gazeStay * gazeScore +
        SIGNAL_WEIGHTS.mouseTrail * mouseScore +
        SIGNAL_WEIGHTS.context * 0 + // 由外部传入当前星系上下文
        SIGNAL_WEIGHTS.heatmap * 0   // 由后端统计提供

      if (!bestCandidate || confidence > bestCandidate.confidence) {
        bestCandidate = { bodyId: body.id, confidence }
      }
    }

    // 触发预加载
    if (bestCandidate && bestCandidate.confidence > 0.6) {
      if (lastPredictionRef.current !== bestCandidate.bodyId) {
        lastPredictionRef.current = bestCandidate.bodyId
        onPrefetch(bestCandidate.bodyId)
      }
    }
  })
}
```

**验证方法**:
1. 视线对准某行星 2 秒 → 该行星数据预加载触发
2. 随意移动视线（不聚焦）→ 不触发预加载
3. 同一目标不重复触发（lastPrediction 判断）
4. 预加载后点击该行星 → 内容瞬间呈现（无加载等待）
